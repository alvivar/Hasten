
// We are trying to simplify the Add with T.

    // Option B
    public static void Add<T>(int id, T obj)
    {
        var type = typeof(T);

        if (type == typeof(EyeTag))
        {
            eyeTagId.Elements[eyeTagId.Length++] = id;
            eyeTag.Elements[eyeTag.Length++] = obj as EyeTag;
        }
    }

    // Option A
    public static void Add<T>(int id, T obj) where T : MonoBehaviour
    {
        var type = typeof(T);
        if (type == typeof(EyeTag))
        {
            eyeTagId.Add(id);
            eyeTag.Add((T) obj);
        }
    }
